<script type="module">
    // ==========================================
    // 1. CONFIGURATION
    // ==========================================
    const CONFIG = {
        TOTAL_ROWS: 150,
        BLOCK_SIZE: 40,
        GAP: 2,
        TOTAL_BLOCKS: 11325,
        // SMOOTH ZOOM CONFIG: Fade in start/end points
        LOD_FADE_START: 0.35,  // Text starts appearing (faintly)
        LOD_FADE_END: 0.60,    // Text becomes fully solid
        BLOCKS_PER_FRAME: 500,    
        API_URL: 'https://your-worker-url.workers.dev', 
        RAZORPAY_KEY: 'rzp_test_YOUR_KEY_HERE',
        TEST_MODE: true 
    };

    const PYRAMID_DIMS = { WIDTH: CONFIG.TOTAL_ROWS * CONFIG.BLOCK_SIZE, HEIGHT: CONFIG.TOTAL_ROWS * CONFIG.BLOCK_SIZE };
    const COLORS = ['#FFD700', '#FF6B6B', '#4ECDC4', '#45B7D1', '#96CEB4', '#FECA57', '#B983FF', '#FD79A8'];
    const FONTS = ['Arial', 'Georgia', 'Courier New', 'Impact', 'Verdana'];

    // ==========================================
    // 2. STATE
    // ==========================================
    let app, container;
    let textures = {}; 
    let blocksData = new Map();
    let gridLookup = [];
    let dirtyBlocks = new Set();

    let state = {
        isDragging: false,
        dragStart: { x: 0, y: 0 },
        appDragStart: { x: 0, y: 0 },
        hoveredBlock: null,
        currentBlock: null,
        minZoom: 0.1
    };

    let formData = resetFormData();

    // ==========================================
    // 3. INIT ENGINE
    // ==========================================
    async function init() {
        app = new PIXI.Application();
        await app.init({
            resizeTo: window,
            backgroundColor: 0x0a0a0f,
            resolution: window.devicePixelRatio || 2,
            autoDensity: true,
            antialias: false,
            powerPreference: 'high-performance'
        });
        document.getElementById('canvas-container').appendChild(app.canvas);

        // INSTALL BITMAP FONT
        PIXI.BitmapFont.install({
            name: 'PyramidFont',
            style: {
                fontFamily: 'Arial',
                fontSize: 64, 
                fontWeight: '900',
                fill: '#ffffff',
                stroke: { color: '#000000', width: 6 },
                align: 'center'
            }
        });

        textures = createTextures(app);
        
        container = new PIXI.Container();
        container.eventMode = 'static'; 
        app.stage.addChild(container);
        
        buildGrid(); 
        calculateMinZoom();
        setupInteraction();
        resetView();
        
        loadData();

        // THE RENDER LOOP
        app.ticker.add(() => {
            processDirtyQueue();      
            updateTextVisibility();   
        });

        setTimeout(() => document.getElementById('loader').remove(), 600);
    }

    // ==========================================
    // 4. GRID BUILDING
    // ==========================================
    function createTextures(app) {
        const makeRect = (fill, stroke, strokeWidth) => {
            const g = new PIXI.Graphics()
                .rect(0, 0, CONFIG.BLOCK_SIZE - CONFIG.GAP, CONFIG.BLOCK_SIZE - CONFIG.GAP)
                .fill(fill)
                .stroke({ width: strokeWidth, color: stroke });
            return app.renderer.generateTexture(g);
        };
        return {
            gold: makeRect(0x2d2d44, 0xFFD700, 3),    
            silver: makeRect(0x252535, 0xC0C0C0, 2),  
            standard: makeRect(0x1e1e28, 0x444444, 1),
            sold: makeRect(0xFFFFFF, 0xFFD700, 3)     
        };
    }

    function buildGrid() {
        let blockId = CONFIG.TOTAL_BLOCKS;
        gridLookup = new Array(CONFIG.TOTAL_ROWS + 1).fill(null).map(() => []);

        for (let row = 1; row <= CONFIG.TOTAL_ROWS; row++) {
            const startX = -(row * CONFIG.BLOCK_SIZE) / 2; 
            const yPos = (row - 1) * CONFIG.BLOCK_SIZE;

            for (let col = 0; col < row; col++) {
                let type = 'standard';
                if (row <= 45) type = 'gold';
                else if (row <= 85) type = 'silver';

                const slot = new PIXI.Sprite(textures[type]);
                slot.x = startX + (col * CONFIG.BLOCK_SIZE);
                slot.y = yPos;
                slot.cullable = true; 

                let price = blockId === CONFIG.TOTAL_BLOCKS ? 5000 : 1 + (blockId - 1) * 0.5;

                const data = { 
                    id: blockId, 
                    price, 
                    tier: type, 
                    row, col, 
                    sold: false, 
                    sprite: slot, 
                    ownerData: null,
                    textObj: null 
                };
                
                blocksData.set(blockId, data);
                gridLookup[row][col] = data;

                container.addChild(slot);
                blockId--;
            }
        }
    }

    // ==========================================
    // 5. ASSEMBLY LINE (QUEUE)
    // ==========================================
    function processDirtyQueue() {
        if (dirtyBlocks.size === 0) return;

        let processed = 0;
        for (const block of dirtyBlocks) {
            if (block.sold) {
                block.sprite.texture = textures.sold;
                block.sprite.tint = block.ownerData.owner_color.replace('#', '0x');
                
                if (!block.textObj) {
                    const textStr = (block.ownerData.owner_text || "HI").substring(0, 10);
                    
                    const text = new PIXI.BitmapText({
                        text: textStr,
                        style: { fontFamily: 'PyramidFont', fontSize: 64, align: 'center' }
                    });
                    
                    text.anchor.set(0.5);
                    text.x = (CONFIG.BLOCK_SIZE - CONFIG.GAP) / 2;
                    text.y = (CONFIG.BLOCK_SIZE - CONFIG.GAP) / 2;
                    const scaleFactor = (CONFIG.BLOCK_SIZE - 10) / text.width;
                    text.scale.set(Math.min(scaleFactor, 0.25)); 
                    
                    text.visible = false; 
                    text.eventMode = 'none'; 
                    text.alpha = 0; // Start invisible
                    
                    block.sprite.addChild(text);
                    block.textObj = text;
                } else {
                    block.textObj.text = (block.ownerData.owner_text || "HI").substring(0, 10);
                }
            } else {
                block.sprite.texture = textures[block.tier];
                block.sprite.tint = 0xFFFFFF;
                if(block.textObj) block.textObj.visible = false;
            }
            
            dirtyBlocks.delete(block);
            processed++;
            if (processed >= CONFIG.BLOCKS_PER_FRAME) break; 
        }
    }

    // ==========================================
    // 6. VISIBILITY TOGGLER (SMOOTH FADE)
    // ==========================================
    function updateTextVisibility() {
        // 1. Calculate Opacity based on Zoom
        const zoom = container.scale.x;
        
        // Linear Interpolation: (Current - Start) / (End - Start)
        let alpha = (zoom - CONFIG.LOD_FADE_START) / (CONFIG.LOD_FADE_END - CONFIG.LOD_FADE_START);
        
        // Clamp between 0.0 and 1.0
        alpha = Math.max(0, Math.min(1, alpha));

        // Optimization: If invisible, don't render at all
        const isVisible = alpha > 0.01;

        // Viewport Culling logic
        const screenTop = -container.y / zoom;
        const screenBottom = (app.screen.height - container.y) / zoom;
        const startRow = Math.max(1, Math.floor(screenTop / CONFIG.BLOCK_SIZE) - 2);
        const endRow = Math.min(CONFIG.TOTAL_ROWS, Math.ceil(screenBottom / CONFIG.BLOCK_SIZE) + 2);

        // Loop visible rows
        for (let r = startRow; r <= endRow; r++) {
            if (!gridLookup[r]) continue;
            const rowData = gridLookup[r];
            for (let c = 0; c < rowData.length; c++) {
                const block = rowData[c];
                
                // Only update if text object exists
                if (block.sold && block.textObj) {
                    block.textObj.visible = isVisible;
                    
                    // Only apply alpha change if visible (saves CPU)
                    if (isVisible) {
                        block.textObj.alpha = alpha; 
                    }
                }
            }
        }
    }

    // ==========================================
    // 7. INTERACTION
    // ==========================================
    function setupInteraction() {
        app.stage.hitArea = app.screen;
        
        app.stage.on('pointerdown', (e) => {
            state.isDragging = true;
            state.dragStart = { x: e.global.x - container.x, y: e.global.y - container.y };
            state.appDragStart = { x: e.global.x, y: e.global.y };
        });
        
        app.stage.on('pointermove', (e) => {
            const globalX = e.global.x;
            const globalY = e.global.y;

            if(state.isDragging) {
                container.x = globalX - state.dragStart.x;
                container.y = globalY - state.dragStart.y;
                constrainPyramid(); 
                return; 
            }

            // Hover Logic
            const localPoint = container.toLocal({x: globalX, y: globalY});
            const r = Math.floor(localPoint.y / CONFIG.BLOCK_SIZE) + 1;
            
            if (r >= 1 && r <= CONFIG.TOTAL_ROWS && gridLookup[r]) {
                const rowWidth = r * CONFIG.BLOCK_SIZE;
                const c = Math.floor((localPoint.x - (-rowWidth/2)) / CONFIG.BLOCK_SIZE);
                const block = gridLookup[r][c];
                
                if (block !== state.hoveredBlock) {
                    if (state.hoveredBlock) state.hoveredBlock.sprite.alpha = 1.0;
                    if (block) {
                        block.sprite.alpha = 0.6; 
                        showTooltip(block, globalX, globalY);
                    } else { hideTooltip(); }
                    state.hoveredBlock = block;
                }
                if (state.hoveredBlock) moveTooltip(globalX, globalY);
            } else {
                if (state.hoveredBlock) state.hoveredBlock.sprite.alpha = 1.0;
                state.hoveredBlock = null;
                hideTooltip();
            }
        });
        
        app.stage.on('pointerup', (e) => {
            if (state.isDragging) {
                state.isDragging = false;
                const dx = e.global.x - state.appDragStart.x;
                const dy = e.global.y - state.appDragStart.y;
                if (Math.sqrt(dx*dx + dy*dy) < 5 && state.hoveredBlock) {
                    state.currentBlock = state.hoveredBlock;
                    openModal();
                }
            }
        });

        // Wheel Zoom
        document.getElementById('canvas-container').addEventListener('wheel', (e) => {
            e.preventDefault();
            const zoomSpeed = 0.0015;
            const mouseX = e.clientX;
            const mouseY = e.clientY;

            const worldPos = {
                x: (mouseX - container.x) / container.scale.x,
                y: (mouseY - container.y) / container.scale.y
            };

            const newScale = Math.min(Math.max(container.scale.x - (e.deltaY * zoomSpeed), state.minZoom), 5.0);
            container.scale.set(newScale);

            const newScreenPos = {
                x: worldPos.x * container.scale.x + container.x,
                y: worldPos.y * container.scale.y + container.y
            };

            container.x -= (newScreenPos.x - mouseX);
            container.y -= (newScreenPos.y - mouseY);
            constrainPyramid();
        }, { passive: false });
    }

    function constrainPyramid() {
        const pyWidth = PYRAMID_DIMS.WIDTH * container.scale.x;
        const pyHeight = PYRAMID_DIMS.HEIGHT * container.scale.y;
        
        if (pyWidth < app.screen.width) container.x = app.screen.width / 2;
        else {
            const minX = app.screen.width - (pyWidth / 2) - 100;
            const maxX = (pyWidth / 2) + 100;
            container.x = Math.max(minX, Math.min(container.x, maxX));
        }
        const minY = app.screen.height - pyHeight - 200;
        const maxY = 200;
        container.y = Math.max(minY, Math.min(container.y, maxY));
    }

    function calculateMinZoom() {
        const scaleX = app.screen.width / (PYRAMID_DIMS.WIDTH + 100); 
        const scaleY = app.screen.height / (PYRAMID_DIMS.HEIGHT + 200);
        state.minZoom = Math.min(scaleX, scaleY);
    }

    // ==========================================
    // 8. DATA LOADING
    // ==========================================
    async function loadData() {
        try {
            let soldSlots = [];
            if (CONFIG.TEST_MODE) { console.log("TEST MODE: Skipping API Fetch"); } 
            else {
                 const res = await fetch(`${CONFIG.API_URL}/api/grid`);
                 if (res.ok) soldSlots = await res.json();
            }
            soldSlots.forEach(slotData => {
                const block = blocksData.get(slotData.slot_number);
                if (block) updateBlockData(block, slotData);
            });
        } catch (e) { console.error("Data Load Error:", e); }
    }

    function updateBlockData(block, serverData) {
        block.sold = true;
        block.ownerData = serverData;
        dirtyBlocks.add(block);
    }

    // ==========================================
    // 9. UI & MODAL LOGIC
    // ==========================================
    window.openModal = () => {
        const modal = document.getElementById('modal-backdrop');
        const content = document.getElementById('modal-content');
        modal.style.display = 'flex';
        content.classList.remove('premium-mode');
        
        if (state.currentBlock.sold) {
            content.classList.add('premium-mode'); 
            renderSoldProfile(content, state.currentBlock);
        } else {
            formData = resetFormData();
            renderPurchaseForm(content, state.currentBlock);
        }
    };

    window.closeModal = (e) => {
        if (!e || e.target.id === 'modal-backdrop' || e.target.classList.contains('close-icon') || e.target.classList.contains('close-icon-floating')) {
            document.getElementById('modal-backdrop').style.display = 'none';
        }
    };

    function resetFormData() {
        return { name: '', message: '', link: '', color: '#FFD700', text: '', font: 'Arial', imageUrl: '' };
    }

    // --- HTML RENDERERS ---
    function renderPurchaseForm(container, block) {

    // --- FORM HANDLERS ---
    window.updateForm = (field, value) => {
        formData[field] = value;
        if (field === 'color') {
            document.getElementById('color-picker').value = value;
            const el = document.getElementById('block-preview');
            el.style.backgroundColor = value;
            el.style.borderColor = value;
        }
        if (field === 'text') {
            const el = document.getElementById('preview-text');
            el.textContent = value || "HI";
            el.style.fontSize = value.length > 5 ? "10px" : "14px";
        }
        if (field === 'font') {
            document.getElementById('preview-text').style.fontFamily = value;
        }
    };

    window.handleImageUpload = async (input) => {
        const file = input.files[0];
        if (!file) return;
        const label = document.getElementById('upload-label');
        label.textContent = "⏳ Uploading...";
        
        if (CONFIG.TEST_MODE) {
            setTimeout(() => { 
                formData.imageUrl = `https://picsum.photos/seed/${Math.random()}/200/200`; 
                label.textContent = "✅ Image Set"; 
            }, 800);
            return;
        }

        const data = new FormData(); 
        data.append('image', file);
        try {
            const res = await fetch(`${CONFIG.API_URL}/api/upload-image`, { method: 'POST', body: data });
            const json = await res.json();
            if (json.success) { 
                formData.imageUrl = json.imageUrl; 
                label.textContent = "✅ Uploaded"; 
            } else { throw new Error("Upload Failed"); }
        } catch (e) { 
            alert('Upload error. Try a smaller image.'); 
            label.textContent = "❌ Error";
        }
    };

    window.initiatePurchase = () => {
        if (!formData.name) { alert("Please enter a name"); return; }
        
        if (CONFIG.TEST_MODE) { 
            finalizePurchase("test_txn_" + Date.now()); 
            return; 
        }

        const options = {
            "key": CONFIG.RAZORPAY_KEY, 
            "amount": state.currentBlock.price * 100, 
            "currency": "USD",
            "name": "History Pyramid", 
            "description": `Block #${state.currentBlock.id}`,
            "handler": (r) => finalizePurchase(r.razorpay_payment_id),
            "prefill": { "name": formData.name }, 
            "theme": { "color": "#FFD700" }
        };
        const rzp = new Razorpay(options);
        rzp.on('payment.failed', (r) => alert("Payment Failed: " + r.error.description));
        rzp.open();
    };

    async function finalizePurchase(paymentId) {
        document.querySelector('.confirm-btn').textContent = "Minting...";
        
        const payload = { 
            slotId: state.currentBlock.id, 
            user: formData.name, 
            message: formData.message, 
            color: formData.color, 
            text: formData.text, 
            font: formData.font,
            paymentId, 
            imageUrl: formData.imageUrl, 
            link: formData.link 
        };

        try {
            if (!CONFIG.TEST_MODE) {
                const res = await fetch(`${CONFIG.API_URL}/api/purchase`, { 
                    method: 'POST', 
                    headers: { 'Content-Type': 'application/json' }, 
                    body: JSON.stringify(payload) 
                });
                if(!res.ok) throw new Error("Server Error");
            }
            
            const serverData = { 
                owner_name: payload.user, 
                owner_color: payload.color, 
                owner_text: payload.text, 
                message: payload.message, 
                imageUrl: payload.imageUrl, 
                link: payload.link,
                font: payload.font
            };
            
            updateBlockData(blocksData.get(state.currentBlock.id), serverData);
            closeModal();
            setTimeout(() => alert(`🎉 Block #${state.currentBlock.id} Minted!`), 100);
            
        } catch (e) { 
            alert("Minting Error: " + e.message); 
            document.querySelector('.confirm-btn').textContent = "Try Again";
        }
    }

    // --- TOOLTIPS ---
    function showTooltip(block, x, y) {
        const t = document.getElementById('tooltip');
        t.style.display = 'block';
        t.textContent = block.sold 
            ? `${block.ownerData.owner_name}` 
            : `#${block.id} • $${block.price.toFixed(2)}`;
        moveTooltip(x, y);
    }
    function moveTooltip(x, y) {
        const t = document.getElementById('tooltip');
        t.style.left = (x + 15) + 'px';
        t.style.top = (y - 35) + 'px';
    }
    function hideTooltip() {
        document.getElementById('tooltip').style.display = 'none';
    }

    };

        if(!confirm("Simulate filling 11,325 blocks? (Stress Test)")) return;

                });

    };
    
    window.viewRandom = () => {
        const sold = Array.from(blocksData.values()).filter(b => b.sold);
        if(sold.length === 0) { alert("No sold blocks yet! Try Simulation Mode."); return; }
        state.currentBlock = sold[Math.floor(Math.random() * sold.length)];
        openModal();
    };
