    // --- 1. CONFIGURATION ---
    const TOTAL_ROWS = 150;
    const BLOCK_SIZE = 40;
    const GAP = 2;
    const TOTAL_BLOCKS = 11325; 
    
    // --- DIMENSIONS & PADDING ---
    const PYRAMID_WIDTH = TOTAL_ROWS * BLOCK_SIZE; 
    const PYRAMID_HEIGHT = TOTAL_ROWS * BLOCK_SIZE; 
    
    // SAFE ZONES (Invisible Walls)
    const BOTTOM_SAFE_ZONE = 120; // 120px space at bottom
    const TOP_HEADER_ZONE = 100;  // 100px space at top
    
    const API_URL = 'https://pyramid-backend.piyushsayare8.workers.dev';
    const RAZORPAY_KEY_ID = 'rzp_test_YOUR_KEY_HERE'; 
    const TEST_MODE = true; 
    
    // OPTIMIZATION: Reduce text resolution on mobile to prevent crashes
    const isMobile = window.innerWidth < 800;
    const TEXT_RESOLUTION = isMobile ? 1.2 : 2.0;

    // --- 2. STATE ---
    let app, container;
    let blocksData = new Map();
    // Optimization: Spatial lookup Grid [row][col] -> data
    let gridLookup = []; 
    let hoveredBlock = null;

    let isDragging = false;
    let dragStart = { x: 0, y: 0 };
    let currentBlock = null;
    let minZoomLevel = 0.1;
    let textures = {}; 
    
    // --- OPTIMIZATION STATE ---
    let isProcessing = false; // Lock: Prevents overlapping simulations/fetches
    let textTextureCache = new Map(); // Cache: Reuses textures for identical text
    
    let formData = { name: '', message: '', link: '', color: '#FFD700', text: '', font: 'Arial', imageUrl: '' };
    const COLORS = ['#FFD700', '#FF6B6B', '#4ECDC4', '#45B7D1', '#96CEB4', '#FECA57', '#B983FF', '#FD79A8'];
    const FONTS = ['Arial', 'Georgia', 'Courier New', 'Comic Sans MS', 'Impact', 'Verdana'];

    // --- 3. PIXIJS ENGINE ---
    async function init() {
        app = new PIXI.Application();
        await app.init({
            resizeTo: window,
            backgroundColor: 0x0a0a0f,
            resolution: window.devicePixelRatio || 2,
            autoDensity: true,
            antialias: true
        });
        document.getElementById('canvas-container').appendChild(app.canvas);

        textures = createTextures(app);
        
        container = new PIXI.Container();
        // Optimization: Container handles events, not individual sprites (Saves 40,000 listeners)
        container.eventMode = 'static'; 
        app.stage.addChild(container);
        
        buildGrid();
        
        calculateMinZoom();
        setupInteraction();
        
        // Start loading real data immediately
        loadData();

    }

    function createTextures(app) {
        const makeRect = (fillColor, strokeColor, width) => {
            const g = new PIXI.Graphics()
                .rect(0, 0, BLOCK_SIZE-GAP, BLOCK_SIZE-GAP)
                .fill(fillColor)
                .stroke({ width: width, color: strokeColor });
            return app.renderer.generateTexture(g);
        };

        return {
            gold: makeRect(0x2d2d44, 0xFFD700, 3),
            silver: makeRect(0x252535, 0xC0C0C0, 2),
            standard: makeRect(0x1e1e28, 0x444444, 1),
            gold_sold: makeRect(0xFFFFFF, 0xFFD700, 3),
            silver_sold: makeRect(0xFFFFFF, 0xC0C0C0, 2),
            standard_sold: makeRect(0xFFFFFF, 0x444444, 1)
        };
    }

    function buildGrid() {
        let blockId = TOTAL_BLOCKS;
        
        // Initialize Grid Lookup Array
        gridLookup = new Array(TOTAL_ROWS + 1).fill(null).map(() => []);

        for (let row = 1; row <= TOTAL_ROWS; row++) {
            const startX = -(row * BLOCK_SIZE) / 2; 
            const yPos = (row - 1) * BLOCK_SIZE;

            for (let col = 0; col < row; col++) {
                let type = 'standard';
                if (row <= 45) type = 'gold';
                else if (row <= 85) type = 'silver';

                const slot = new PIXI.Sprite(textures[type]);
                slot.x = startX + (col * BLOCK_SIZE);
                slot.y = yPos;
                
                let price = blockId === TOTAL_BLOCKS ? 5000 : 1 + (blockId - 1) * 0.5;

                const data = { id: blockId, price, tier: type, row, col, sold: false, sprite: slot };
                
                // Store in both Map (for ID access) and Grid (for click access)
                blocksData.set(blockId, data);
                gridLookup[row][col] = data;

                container.addChild(slot);
                blockId--;
            }
        }
    }

    // --- 4. OPTIMIZED VISUAL ENGINE (The "Engine Room") ---
    function updateSlotVisuals(localData, serverData) {
        localData.sold = true; 
        localData.ownerData = serverData;
        
        const soldTextureKey = localData.tier + '_sold';
        localData.sprite.texture = textures[soldTextureKey];
        localData.sprite.tint = serverData.owner_color.replace('#', '0x');
        
        // Remove existing text to prevent leaks
        if (localData.sprite.children.length > 0) {
            localData.sprite.removeChildren();
        }

        if (serverData.owner_text) {
            const textStr = serverData.owner_text.substring(0, 80);
            
            // --- CRITICAL PERFORMANCE UPGRADE: TEXTURE CACHING ---
            // 1. Check if we have already drawn this EXACT word.
            // 2. If yes, reuse the GPU texture. (0ms cost)
            // 3. If no, create it once and save it. (5ms cost)
            
            let targetTexture = textTextureCache.get(textStr);

            if (!targetTexture) {
                // Generate Texture (CPU Heavy - Done only once per unique phrase)
                const VIRTUAL_FONT_SIZE = 48; 
                const APPROX_CHAR_WIDTH = VIRTUAL_FONT_SIZE * 0.7; 
                const idealSquareSide = Math.sqrt(textStr.length) * APPROX_CHAR_WIDTH;
                const MIN_WIDTH = VIRTUAL_FONT_SIZE * 5; 
                const targetWrapWidth = Math.max(idealSquareSide, MIN_WIDTH);

                const style = new PIXI.TextStyle({ 
                    fontFamily: 'Arial', fontSize: VIRTUAL_FONT_SIZE, fontWeight: '900', 
                    fill: '#ffffff', stroke: { color: '#000000', width: 6 }, 
                    dropShadow: true, dropShadowColor: '#000000', dropShadowBlur: 3, dropShadowDistance: 2,
                    align: 'center', wordWrap: true, wordWrapWidth: targetWrapWidth, lineHeight: VIRTUAL_FONT_SIZE * 1.1, padding: 10
                });

                const tempText = new PIXI.Text({ text: textStr, style });
                tempText.resolution = TEXT_RESOLUTION; // Optimized resolution
                
                targetTexture = app.renderer.generateTexture(tempText);
                
                // Limit Cache Size to prevent memory overflow (keep last 1000 unique phrases)
                if(textTextureCache.size > 1000) {
                    const firstKey = textTextureCache.keys().next().value;
                    textTextureCache.delete(firstKey); 
                }
                textTextureCache.set(textStr, targetTexture);
            }

            // Create Sprite from Texture (GPU Light)
            const textSprite = new PIXI.Sprite(targetTexture);
            textSprite.anchor.set(0.5);

            // Calculate scaling to fit the box
            const BOX_SIZE = BLOCK_SIZE - GAP; 
            const SAFE_ZONE = 34; 
            const scaleX = SAFE_ZONE / textSprite.width;
            const scaleY = SAFE_ZONE / textSprite.height;
            const finalScale = Math.min(scaleX, scaleY); 

            textSprite.scale.set(finalScale);
            textSprite.x = BOX_SIZE / 2; 
            textSprite.y = BOX_SIZE / 2;
            
            localData.sprite.addChild(textSprite);
        }
    }

    // --- 5. INTERACTION LOGIC (Spatial Hashing) ---
    function calculateMinZoom() {
        const scaleX = app.screen.width / (PYRAMID_WIDTH + 200); 
        const scaleY = app.screen.height / (PYRAMID_HEIGHT + 300);
        minZoomLevel = Math.min(scaleX, scaleY);
    }

    function constrainPyramid() {
        if (container.scale.x < minZoomLevel) container.scale.set(minZoomLevel);

        const pyWidth = PYRAMID_WIDTH * container.scale.x;
        const pyHeight = PYRAMID_HEIGHT * container.scale.y;

        if (pyWidth < app.screen.width) {
            container.x = app.screen.width / 2;
        } else {
            const minX = app.screen.width - (pyWidth / 2) - 100;
            const maxX = (pyWidth / 2) + 100;
            if (container.x < minX) container.x = minX;
            if (container.x > maxX) container.x = maxX;
        }

        if (pyHeight < app.screen.height) {
            container.y = (app.screen.height - pyHeight) / 2;
        } else {
            const maxY = TOP_HEADER_ZONE; 
            const minY = app.screen.height - pyHeight - BOTTOM_SAFE_ZONE;
            if (container.y < minY) container.y = minY;
            if (container.y > maxY) container.y = maxY;
        }
    }

    // O(1) Lookup - No Loops
    function getBlockAtGlobalPos(globalX, globalY) {
        const localPoint = container.toLocal({x: globalX, y: globalY});
        
        // Grid Math
        const r = Math.floor(localPoint.y / BLOCK_SIZE) + 1;
        if (r < 1 || r > TOTAL_ROWS) return null;
        
        const rowWidth = r * BLOCK_SIZE;
        const rowStartX = -rowWidth / 2;
        const c = Math.floor((localPoint.x - rowStartX) / BLOCK_SIZE);
        if (c < 0 || c >= r) return null;
        
        return gridLookup[r] ? gridLookup[r][c] : null;
    }

    function setupInteraction() {
        app.stage.eventMode = 'static';
        app.stage.hitArea = app.screen;
        
        // Pan Logic
        app.stage.on('pointerdown', (e) => {
            isDragging = true;
            dragStart = { x: e.global.x - container.x, y: e.global.y - container.y };
            app.dragStartPos = { x: e.global.x, y: e.global.y }; 
        });
        
        app.stage.on('pointermove', (e) => {
            const globalX = e.global.x;
            const globalY = e.global.y;

            if(isDragging) {
                container.x = globalX - dragStart.x;
                container.y = globalY - dragStart.y;
                constrainPyramid();
                return; 
            }

            const targetBlock = getBlockAtGlobalPos(globalX, globalY);

            if (targetBlock !== hoveredBlock) {
                if (hoveredBlock) onHoverOut(hoveredBlock);
                if (targetBlock) onHover(globalX, globalY, targetBlock);
                hoveredBlock = targetBlock;
            } else if (targetBlock) {
                updateTooltipPos(globalX, globalY);
            }
        });
        
        app.stage.on('pointerup', (e) => {
            if (isDragging) {
                isDragging = false;
                
                const dx = e.global.x - (app.dragStartPos?.x || 0);
                const dy = e.global.y - (app.dragStartPos?.y || 0);
                
                if (Math.sqrt(dx*dx + dy*dy) < 5) {
                   const targetBlock = getBlockAtGlobalPos(e.global.x, e.global.y);
                   if (targetBlock) {
                       currentBlock = targetBlock;
                       openModal();
                   }
                }
            }
        });
        
        app.stage.on('pointerupoutside', () => isDragging = false);

        // Wheel Zoom
        document.getElementById('canvas-container').addEventListener('wheel', (e) => {
            e.preventDefault();
            const zoomSpeed = 0.001;
            const mouseX = e.clientX;
            const mouseY = e.clientY;

            const worldPos = {
                x: (mouseX - container.x) / container.scale.x,
                y: (mouseY - container.y) / container.scale.y
            };

            const newScale = container.scale.x - (e.deltaY * zoomSpeed);
            const clampedScale = Math.min(Math.max(newScale, minZoomLevel), 4.0);
            container.scale.set(clampedScale);

            const newScreenPos = {
                x: worldPos.x * container.scale.x + container.x,
                y: worldPos.y * container.scale.y + container.y
            };

            container.x -= (newScreenPos.x - mouseX);
            container.y -= (newScreenPos.y - mouseY);
            
            constrainPyramid();
        }, { passive: false });
    }

    // --- 6. VISUAL FEEDBACK (Tooltip) ---

        updateTooltipPos(gx, gy);
    }

        const tooltip = document.getElementById('tooltip');
        tooltip.style.left = (gx + 15) + 'px';
        tooltip.style.top = (gy - 30) + 'px';
    }

    function onHoverOut(data) {
        if(data && data.sprite) data.sprite.alpha = 1.0;
        document.getElementById('tooltip').style.display = 'none';
    }

    // --- OPTIMIZED: CLOUDFLARE/SERVER DATA LOADING ---
    async function loadData() {
        if(isProcessing) return; // Prevent double loads
        isProcessing = true;
        
        try {
            const res = await fetch(`${API_URL}/api/grid`);
            if (!res.ok) throw new Error();
            
            const soldSlots = await res.json();
            
            // Chunking: Process 40 items per frame (Safe for production)
            let index = 0;
            const batchSize = 40;

            function processBatch() {
                const limit = Math.min(index + batchSize, soldSlots.length);
                for (let i = index; i < limit; i++) {
                    const slotData = soldSlots[i];
                    const localBlock = blocksData.get(slotData.slot_number);
                    if (localBlock) {
                        updateSlotVisuals(localBlock, slotData);
                    }
                }
                index += batchSize;
                
                if (index < soldSlots.length) {
                    requestAnimationFrame(processBatch);
                } else {
                    isProcessing = false; // Release lock
                }
            }
            processBatch();
            
        } catch (e) { 
            console.log("Simulating offline mode / Error loading data"); 
            isProcessing = false;
        }
    }

            }
            index += batchSize;
            
            if (index < candidates.length) {
                requestAnimationFrame(processBatch);
            } else {
                isProcessing = false; // Unlock
            }
        }
        processBatch();
    };
